图的遍历：BFS与DFS

2026.04.19 · C++竞赛 · 约 7 分钟阅读

图的遍历是图论算法的基础，BFS（广度优先搜索）和DFS（深度优先搜索）是两种最基本的遍历方式，在竞赛中应用广泛。

图的存储结构

在实现遍历之前，首先需要选择合适的图存储方式。常用的有邻接矩阵和邻接表。

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1005;

// 邻接表存储
vector<int> adj[MAXN];

// 添加边（无向图）
void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

// 初始化图
void initGraph(int n) {
    for (int i = 0; i < n; i++) {
        adj[i].clear();
    }
}

深度优先搜索（DFS）

DFS沿着一条路径尽可能深地搜索，直到无法继续才回溯。适合解决连通性、路径搜索等问题。

bool visited[MAXN];  // 访问标记数组

// DFS递归实现
void dfs(int node) {
    visited[node] = true;
    cout << node << " ";  // 处理当前节点
    
    for (int neighbor : adj[node]) {
        if (!visited[neighbor]) {
            dfs(neighbor);
        }
    }
}

// 计算连通分量个数
int countComponents(int n) {
    int components = 0;
    
    for (int i = 0; i < n; i++) {
        visited[i] = false;
    }
    
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfs(i);
            components++;
            cout << endl;
        }
    }
    
    return components;
}

int main() {
    int n, m;
    cin >> n >> m;
    
    initGraph(n);
    
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        addEdge(u, v);
    }
    
    cout << "连通分量个数: " << countComponents(n) << endl;
    
    return 0;
}

广度优先搜索（BFS）

BFS逐层扩展，先访问距离起点近的节点。常用于求最短路径、层次遍历等问题。

int dist[MAXN];  // 距离数组

// BFS求最短路径
void bfs(int start, int n) {
    for (int i = 0; i < n; i++) {
        dist[i] = -1;  // -1表示未访问
    }
    
    queue<int> q;
    q.push(start);
    dist[start] = 0;
    
    while (!q.empty()) {
        int node = q.front();
        q.pop();
        
        for (int neighbor : adj[node]) {
            if (dist[neighbor] == -1) {
                dist[neighbor] = dist[node] + 1;
                q.push(neighbor);
            }
        }
    }
}

// 打印最短距离
void printDistances(int n) {
    for (int i = 0; i < n; i++) {
        if (dist[i] == -1) {
            cout << "节点 " << i << ": 不可达" << endl;
        } else {
            cout << "节点 " << i << ": 距离为 " << dist[i] << endl;
        }
    }
}

实际应用示例

下面是一个使用BFS解决迷宫最短路径问题的完整示例：

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 105;
char maze[MAXN][MAXN];
int dist[MAXN][MAXN];
int dx[] = {0, 0, 1, -1};  // 四个方向
int dy[] = {1, -1, 0, 0};

struct Point {
    int x, y;
};

int bfsMaze(int startX, int startY, int endX, int endY, int n, int m) {
    memset(dist, -1, sizeof(dist));
    
    queue<Point> q;
    q.push({startX, startY});
    dist[startX][startY] = 0;
    
    while (!q.empty()) {
        Point curr = q.front();
        q.pop();
        
        // 到达终点
        if (curr.x == endX && curr.y == endY) {
            return dist[endX][endY];
        }
        
        // 探索四个方向
        for (int i = 0; i < 4; i++) {
            int nx = curr.x + dx[i];
            int ny = curr.y + dy[i];
            
            // 检查边界和障碍物
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && 
                maze[nx][ny] != '#' && dist[nx][ny] == -1) {
                dist[nx][ny] = dist[curr.x][curr.y] + 1;
                q.push({nx, ny});
            }
        }
    }
    
    return -1;  // 无法到达
}

int main() {
    int n, m;
    cin >> n >> m;
    
    int startX, startY, endX, endY;
    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> maze[i][j];
            if (maze[i][j] == 'S') {
                startX = i;
                startY = j;
            } else if (maze[i][j] == 'T') {
                endX = i;
                endY = j;
            }
        }
    }
    
    int result = bfsMaze(startX, startY, endX, endY, n, m);
    
    if (result == -1) {
        cout << "无法到达终点" << endl;
    } else {
        cout << "最短路径长度: " << result << endl;
    }
    
    return 0;
}

算法对比

  • DFS：空间复杂度低，适合判断连通性、拓扑排序、寻找所有路径
  • BFS：能找到无权图的最短路径，适合层次遍历、最小步数问题
  • 时间复杂度：两者都是O(V+E)，其中V是顶点数，E是边数

选择BFS还是DFS取决于具体问题。记住：求最短路径用BFS，判断连通性或枚举所有可能用DFS。

熟练掌握这两种遍历方式是学习更高级图论算法（如Dijkstra、Floyd、Tarjan等）的基础，建议多做相关练习题目。
